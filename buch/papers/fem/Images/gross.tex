\def\xstep{0.145}
\def\ystep{-0.145}

\def\quadrat#1#2#3{
	\fill[color=#3,opacity=0.2]
		({({#1}-0.5)*\xstep},{({#2}-0.5)*\ystep})
		rectangle
		({({#1}+0.5)*\xstep},{({#2}+0.5)*\ystep});
}

\def\oberesdreieck#1{
	\quadrat{#1}{#1}{red}
	\quadrat{#1+6}{#1}{red}
	\quadrat{#1+7}{#1}{red}
	\quadrat{#1}{#1+6}{red}
	\quadrat{#1+6}{#1+6}{red}
	\quadrat{#1+7}{#1+6}{red}
	\quadrat{#1}{#1+7}{red}
	\quadrat{#1+6}{#1+7}{red}
	\quadrat{#1+7}{#1+7}{red}
}

\def\unteresdreieck#1{
	\quadrat{#1}{#1}{blue}
	\quadrat{#1+1}{#1}{blue}
	\quadrat{#1+7}{#1}{blue}
	\quadrat{#1}{#1+1}{blue}
	\quadrat{#1+1}{#1+1}{blue}
	\quadrat{#1+7}{#1+1}{blue}
	\quadrat{#1}{#1+7}{blue}
	\quadrat{#1+1}{#1+7}{blue}
	\quadrat{#1+7}{#1+7}{blue}
}

\begin{scope}[xshift=7cm,yshift=5cm]

	\foreach \j in {0,...,4}{
